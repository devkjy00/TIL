
### 스택을 사용하여 후위 표기식으로 변환하는 알고리즘
    infix_to_postfix(exp) 
        while(true) do {
            symbol <- getSymbol(exp); 
            case {
             symbol = operand :    // 피연산자 처리
                print(symbol); 
            symbol = operator :    // 연산자 처리
                push(stack, symbol);
            symbol = “ ) ” :       // 오른쪽 괄호 처리
                print(pop(stack));
            symbol = null :        // 수식의 끝 처리
                while (top > -1) do
                    print(pop(stack));
            else : 
        }
    }
    end infix_to_postfix( )
